    ## Installation

    ```npm
    npx shadcn@latest add @dotui/color-swatch
    ```

    ## Usage

    Use color swatches to display a color sample.

    ```tsx
    import { ColorSwatch } from "@/components/ui/color-swatch";
    ```

    ```tsx
    <ColorSwatch color="#ff0000" />
    ```
