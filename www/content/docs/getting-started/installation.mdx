### Install Tailwind CSS:

Install `tailwindcss` and its peer dependencies, then generate your `tailwind.config.ts` and `postcss.config.js` files.

```package-install
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init --ts
```

Configure your tailwind template paths

```ts title="tailwind.config.ts"
import type { Config } from "tailwindcss";

export default {
  content: ["./index.html", "./src/**/*.{ts,tsx,js,jsx}"],
  theme: {
    extend: {},
  },
  plugins: [],
} satisfies Config;
```

Add the Tailwind directives to your CSS

```css title="index.css"
@tailwind base;
@tailwind components;
@tailwind utilities;
/* ... */
```

<Step>Add import alias support:</Step>

```package-install
npm install -D tsconfig-paths
```

Update these files:

Configure your tailwind template paths

```ts title="tailwind.config.ts"
import type { Config } from "tailwindcss";

export default {
  content: ["./app/**/{**,.client,.server}/**/*.{js,jsx,ts,tsx}"],
  theme: {
    extend: {},
  },
  plugins: [],
} satisfies Config;
```

Add the Tailwind directives to your CSS

```css title="app/tailwind.css"
@tailwind base;
@tailwind components;
@tailwind utilities;
/* ... */
```

Then add `tailwind.css` to your root route's links function:

```tsx title="app/root.tsx"
import styles from "./tailwind.css?url";

export const links: LinksFunction = () => [{ rel: "stylesheet", href: styles }];
```

## Manual installation

<Steps>

### Add Tailwind CSS to your project.

Follow the [Tailwind CSS installation guide](https://tailwindcss.com/docs/installation).

### Configure the import alias

```ts title="tsconfig.json"
{
  "compilerOptions": {
    // ...
    "baseUrl": ".",
    "paths": {
      "@/*": ["./src/*"]
    }
    // ...
  }
}
```

### Add required dependencies

```package-install
npm install tailwind-variants clsx tailwind-merge tailwindcss-animate tailwindcss-react-aria-components
```

### Add an icon library

If you're using the default icon library, install `lucide-react`:

```package-install
npm install lucide-react
```

### Add helper functions

```ts title="lib/cn.ts"
import { clsx, type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```
