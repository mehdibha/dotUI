## Installation

```npm
npx shadcn@latest add @dotui/skeleton
```

## Usage

Use skeleton components as placeholders while content is loading.

```tsx
import { Skeleton } from "@/components/ui/skeleton";
```

```tsx
<Skeleton isLoading={true}>
	<Skeleton className="h-4 w-32" />
</Skeleton>
```
