    ## Installation

    ```npm
    npx shadcn@latest add @dotui/slider
    ```

    ## Usage

    Use slider to allow users to select a value from within a given range.

    ```tsx
    import { Slider, SliderControl } from "@/components/ui/slider";
    ```

    ```tsx
    <Slider defaultValue={50}>
      <SliderControl />
    </Slider>
    ```
