## Installation

```npm
npx shadcn@latest add @dotui/field
```

## Usage

```tsx
import { Field, Label, Description, FieldError } from "@/components/ui/field";
```

```tsx
<Field>
  <Label>Email</Label>
  <Input type="email" />
  <Description>We'll never share your email.</Description>
</Field>
```
