    ## Installation

    ```npm
    npx shadcn@latest add @dotui/dialog
    ```

    ## Usage

    Use dialog to grab the user's attention for critical tasks, like confirming actions, providing additional details, or capturing input.

    ```tsx
    import { Button } from "@/components/ui/button";
    import {
      Dialog,
      DialogBody,
      DialogContent,
      DialogDescription,
      DialogFooter,
      DialogHeader,
      DialogHeading,
    } from "@/components/ui/dialog";
    import { Overlay } from "@/components/ui/overlay";
    ```

    ```tsx
    <Dialog>
      <Button>Open dialog</Button>
      <Overlay>
        <DialogContent>
          <DialogHeader>
            <DialogHeading>Dialog Title</DialogHeading>
            <DialogDescription>Dialog description</DialogDescription>
          </DialogHeader>
          <DialogBody>Content</DialogBody>
          <DialogFooter>
            <Button slot="close">Close</Button>
          </DialogFooter>
        </DialogContent>
      </Overlay>
    </Dialog>
    ```
