    ## Installation

    ```npm
    npx shadcn@latest add @dotui/badge
    ```

    ## Usage

    Use badges to highlight status, categorize items, or show counts.

    ```tsx
    import { Badge } from "@/components/ui/badge";
    ```

    ```tsx
    <Badge>Badge</Badge>
    ```
