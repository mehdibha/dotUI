    ## Installation

    ```npm
    npx shadcn@latest add @dotui/calendar
    ```

    ## Usage

    Use calendars to allow users to select a single date value.

    ```tsx
    import { Calendar } from "@/components/ui/calendar";
    ```

    ```tsx
    <Calendar aria-label="Date" />
    ```
