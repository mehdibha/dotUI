## Installation

```npm
npx shadcn@latest add @dotui/card
```

## Usage

```tsx
import { Card, CardHeader, CardContent, CardFooter } from "@/components/ui/card";
```

```tsx
<Card>
  <CardHeader>Title</CardHeader>
  <CardContent>Content goes here.</CardContent>
  <CardFooter>Footer</CardFooter>
</Card>
```
