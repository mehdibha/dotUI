## Installation

```npm
npx shadcn@latest add @dotui/input
```

## Usage

```tsx
import { InputGroup, InputAddon } from "@/components/ui/input";
```

```tsx
<InputGroup>
  <InputAddon>https://</InputAddon>
  <Input placeholder="example.com" />
</InputGroup>
```
