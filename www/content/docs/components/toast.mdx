## Installation

```npm
npx shadcn@latest add @dotui/toast
```

## Usage

```tsx
import { ToastProvider, toastManager } from '@/components/ui/toast'
```

```tsx
;<ToastProvider />

toastManager.add({
  title: 'Your message has been sent.',
})

toastManager.add({
  title: 'Changes saved',
  description: 'Your update is live.',
  type: 'success',
})
```
