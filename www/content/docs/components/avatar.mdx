    ## Installation

    ```npm
    npx shadcn@latest add @dotui/avatar
    ```

    ## Usage

    Use avatars to represent users or entities with an image or initials.

    ```tsx
    import { Avatar } from "@/components/ui/avatar";
    ```

    ```tsx
    <Avatar
      src="https://github.com/mehdibha.png"
      alt="@mehdibha"
      fallback="MB"
    />
    ```
