## Installation

```npm
npx shadcn@latest add @dotui/tag-group
```

## Usage

```tsx
import { TagGroup, TagList, Tag } from '@/components/ui/tag-group'
```

```tsx
<TagGroup>
  <TagList>
    <Tag>React</Tag>
    <Tag>TypeScript</Tag>
  </TagList>
</TagGroup>
```
