    ## Installation

    ```npm
    npx shadcn@latest add @dotui/color-picker
    ```

    ## Usage

    Use color pickers to allow users to select a color from a palette or input a custom color value.

    ```tsx
    import { ColorEditor } from "@/components/ui/color-editor";
    import { ColorPicker, ColorPickerContent, ColorPickerTrigger } from "@/components/ui/color-picker";
    ```

    ```tsx
    <ColorPicker defaultValue="#ff0000">
      <ColorPickerTrigger />
      <ColorPickerContent>
        <ColorEditor />
      </ColorPickerContent>
    </ColorPicker>
    ```

    ## API Reference

    | Prop           | Type        | Default | Description                              |
    | -------------- | ----------- | ------- | ---------------------------------------- |
    | `value`        | `Color`     | -       | The current value (controlled).          |
    | `defaultValue` | `Color`     | -       | The default value (uncontrolled).        |
    | `isDisabled`   | `boolean`   | -       | Whether the color picker is disabled.    |
    | `children`     | `ReactNode` | -       | The color picker trigger and content.    |

    | Event      | Type                     | Description                              |
    | ---------- | ------------------------ | ---------------------------------------- |
    | `onChange` | `(value: Color) => void` | Handler that is called when the value changes. |
