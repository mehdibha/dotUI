    ## Installation

    ```npm
    npx shadcn@latest add @dotui/file-trigger
    ```

    ## Usage

    Use file triggers to allow users to upload files using any pressable element.

    ```tsx
    import { Button } from "@/components/ui/button";
    import { FileTrigger } from "@/components/ui/file-trigger";
    ```

    ```tsx
    <FileTrigger onSelect={(e) => console.log(e)}>
      <Button>Upload File</Button>
    </FileTrigger>
    ```

    ## API Reference

    | Prop              | Type                          | Default | Description                                      |
    | ----------------- | ----------------------------- | ------- | ------------------------------------------------ |
    | `acceptedFileTypes` | `string[]`                  | -       | The types of files the input should accept.      |
    | `allowsMultiple`  | `boolean`                     | -       | Whether multiple files can be selected.          |
    | `defaultCamera`   | `'user' \| 'environment'`     | -       | The camera to use for capturing media.           |
    | `acceptDirectory` | `boolean`                     | -       | Whether to allow directory selection.            |
    | `children`        | `ReactNode`                   | -       | The trigger element.                             |

    | Event      | Type                               | Description                                      |
    | ---------- | ---------------------------------- | ------------------------------------------------ |
    | `onSelect` | `(files: FileList \| null) => void`| Handler that is called when files are selected.  |
