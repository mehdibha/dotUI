    ## Installation

    ```npm
    npx shadcn@latest add @dotui/color-field
    ```

    ## Usage

    Use color fields to allow users to edit a color value using a text field.

    ```tsx
    import { ColorField } from "@/components/ui/color-field";
    import { Label } from "@/components/ui/field";
    import { Input } from "@/components/ui/input";
    ```

    ```tsx
    <ColorField defaultValue="#ff0000">
      <Label>Color</Label>
      <Input />
    </ColorField>
    ```

    ## API Reference

    | Prop           | Type        | Default | Description                              |
    | -------------- | ----------- | ------- | ---------------------------------------- |
    | `value`        | `Color`     | -       | The current value (controlled).          |
    | `defaultValue` | `Color`     | -       | The default value (uncontrolled).        |
    | `isDisabled`   | `boolean`   | -       | Whether the color field is disabled.     |
    | `isReadOnly`   | `boolean`   | -       | Whether the color field is read only.    |
    | `isInvalid`    | `boolean`   | -       | Whether the current value is invalid.    |
    | `isRequired`   | `boolean`   | -       | Whether the field is required.           |
    | `channel`      | `ColorChannel` | -    | The color channel that the field edits.  |

    | Event      | Type                     | Description                              |
    | ---------- | ------------------------ | ---------------------------------------- |
    | `onChange` | `(value: Color) => void` | Handler that is called when the value changes. |
