## Installation

```npm
npx shadcn@latest add @dotui/accordion
```

## Anatomy

```tsx
import {
  Accordion,
  AccordionItem,
  AccordionPanel,
  AccordionTrigger,
} from '@/components/ui/accordion'
```

```tsx
<Accordion>
  <AccordionItem>
    <AccordionTrigger>Accordion Heading</AccordionTrigger>
    <AccordionPanel>Accordion Panel</AccordionPanel>
  </AccordionItem>
  <AccordionItem>
    <AccordionTrigger>Accordion Heading</AccordionTrigger>
    <AccordionPanel>Accordion Panel</AccordionPanel>
  </AccordionItem>
</Accordion>
```
