## Installation

```npm
npx shadcn@latest add @dotui/scroll-fade
```

## Usage

```tsx
import { ScrollFade } from '@/components/ui/scroll-fade'
```

```tsx
<ScrollFade className="h-64">
  <div>Scrollable content</div>
</ScrollFade>
```
